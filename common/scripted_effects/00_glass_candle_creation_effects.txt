create_artifact_glass_candle_black_effect = {
	# Get the character the artifact is being made for.
	# Not really used, but if we don't set the scopes we get errors in the feature selection
	set_artifact_rarity_illustrious = yes

	# Create the artifact
	create_artifact = {
		name = artifact_glass_candle_name
		description = artifact_glass_candle_desc
		template = general_unique_template
		type = glass_candle
		visuals = glass_candle_black 
		wealth = 20
		quality = 20
		history = {
			type = discovered
			actor = scope:giver
			recipient = scope:owner
			location = province:6620
		}
		modifier = artifact_court_grandeur_baseline_add_1_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		hidden_effect = {
			get_random_artifact_modifier_misc_low_effect = yes
		}
		set_variable = { name = historical_unique_artifact value = yes }
		save_scope_as = epic
	}
}
