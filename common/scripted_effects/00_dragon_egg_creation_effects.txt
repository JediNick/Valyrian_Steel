create_artifact_dragon_egg_blue_effect_vs = {

	set_artifact_rarity_illustrious = yes

	# Create the artifact
	create_artifact = {
		name = dragon_egg_name
		description = dragon_egg_description
		template = general_unique_template
		type = dragon_egg
		visuals = dragon_egg_blue
		wealth = 30
		quality = 30
		history = {
			type = discovered
			actor = scope:giver
			recipient = scope:owner
			location = province:6620
		}
		modifier = artifact_dynasty_opinion_3_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		hidden_effect = {
			get_random_artifact_modifier_misc_effect = yes
		}
		set_variable = { name = historical_unique_artifact value = yes }
		set_variable = blue_egg
		set_variable = {
 			name = dragon_egg
 			value = yes
 		}
		save_scope_as = epic
	}
}
