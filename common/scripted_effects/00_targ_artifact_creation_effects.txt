create_artifact_vs_sword_1_effect = {
	set_artifact_rarity_illustrious = yes

	create_artifact = {
		name = artifact_vs_sword_1_name
		description = artifact_vs_sword_1_description
		template = general_unique_template
		type = sword
		visuals = vs_sword_1
		wealth = 50
		quality = 50
		history = {
			type = discovered
			actor = scope:giver
			recipient = scope:owner
			location = province:6620
		}

		modifier = artifact_prowess_1_modifier
		save_scope_as = newly_created_artifact
		decaying = no
	}
	scope:newly_created_artifact = {
		get_random_artifact_modifier_effect = yes
		set_variable = { name = historical_unique_artifact value = yes }
		set_variable = {
			name = valyrian_steel
			value = yes
		}
		set_variable = vs_sword_1
		save_scope_as = epic
	}
}

get_random_valyrian_steel_artifact = {
	$OWNER$ = { save_scope_as = owner }
	$GIVER$ = { save_scope_as = giver }
	random_list = {
		1 = {
			create_artifact_vs_axe_effect = yes
		}
		1 = {
			create_artifact_vs_mace_effect = yes
		}
		1 = {
			create_artifact_vs_spear_1_effect = yes
		}
		1 = {
			create_artifact_vs_dagger_1_effect = yes
		}
		1 = {
			create_artifact_vs_sword_1_effect = yes
		}
		1 = {
			create_artifact_vs_sword_2_effect = yes
		}
		1 = {
			create_artifact_vs_sword_3_effect = yes
		}
		1 = {
			create_artifact_vs_sword_4_effect = yes
		}
	}
}

get_default_valyrian_steel_effect = {
	if = {
		limit = {
			scope:newly_created_artifact = {
				has_variable = vs_axe
			}
		}
		add_character_flag = vs_axe_default
	}
	else_if = {
		limit = {
			scope:newly_created_artifact = {
				OR = {
					has_variable = vs_sword_1
					has_variable = vs_sword_2
					has_variable = vs_sword_3
				}
			}
		}
		add_character_flag = vs_sword_default
	}
	else_if = {
		limit = {
			scope:newly_created_artifact = {
				has_variable = vs_dagger_1
			}
		}
		add_character_flag = vs_dagger_default
	}
	else_if = {
		limit = {
			scope:newly_created_artifact = {
				has_variable = vs_spear_1
			}
		}
		add_character_flag = vs_spear_default
	}
	else_if = {
		limit = {
			scope:newly_created_artifact = {
				has_variable = vs_mace
			}
		}
		add_character_flag = vs_mace_default
	}
	else = {
		add_character_flag = vs_default
	}
}
